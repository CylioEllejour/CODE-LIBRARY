#include <bits/stdc++.h>

using namespace std;

const int Max = 2e+5;

struct data
{
    int val;
    int id;
} num[Max];

int newa[Max];

bool cmp(data x, data y)
{
    return x.val < y.val;
}

int main()
{
    int t;
    cin >> t;
    while (t--)
    {
        memset(newa,sizeof(newa),0);
        int n;
        cin >> n;
        for (int i = 1; i <= n; i++)
        {
            cin >> num[i].val;
            num[i].id = i;
        }
        sort(num + 1, num + 1 + n, cmp);
        for(int i=1;i<=n;i++)
        {
            newa[num[i].id]=i;

            // if(num[i].val==num[i-1].val)
            // newa[num[i].id] = newa[num[i-1].id];
        }
        for(int i=n;i>0;i--)
        cout<<newa[i]<<" ";
        cout<<endl;
    }

    return 0;
}
