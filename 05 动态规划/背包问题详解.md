# 各背包问题对比分析

## 01背包

### 朴素做法

DP本质上还是暴力。但对比纯暴力做法，DP相当与建立了一个表格，将有着共同结论的一类放入表格，做到极大的优化查询。

而从其核心部分**状态转移方程**和**最优子结构**来看，其答案如同在上一层构建的全局解中构建下一层的全局解。

```C++
#include <bits/stdc++.h>

using namespace std;

const int N = 1010;

int w[N], v[N];
int n, m;

int dp[N];
int main()
{
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
    {
        cin >> v[i];
        cin >> w[i];
    }
    for (int i = 1; i <= n; i++)
        for (int j = m; j >= 0; j--)
        {
            dp[j] = dp[j];
            if (j >= v[i])
                dp[j] = max(dp[j], dp[j - v[i]] + w[i]);
        }
    cout << dp[m] << endl;
    return 0;
}
```

### 滚动数组优化

```C++
#include <bits/stdc++.h>

using namespace std;

const int N = 1010;

int w[N], v[N];
int n, m;

int dp[N][N];
int main()
{
    cin>>n>>m;
    for(int i =1;i<=n;i++)
    {
        cin>>v[i]>>w[i];
    }
    for(int i = 1;i<=n;i++)
    {
        for(int j = 0;j <= m;j++)
        {
            dp[i][j] = dp[i-1][j];
            if(j >= v[i])
            dp[i][j] = max(dp[i][j],dp[i][j-v[i]] + w[i]);
        }
    }
    cout<<dp[n][m]<<endl;
    return 0;
}
```

## 分组背包

朴素做法

```C++
#include <bits/stdc++.h>

using namespace std;

const int N = 1010;

int w[N][N], v[N][N], s[N];
int n, m;

int dp[N][N];
int main()
{
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
    {
        cin >> s[i];
        for (int j = 1; j <= s[i]; j++)
        {
            cin >> v[i][j] >> w[i][j];
        }
    }

    for (int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= m; j++)
        {
            dp[i][j] = dp[i - 1][j];
            for (int k = 0; k <= s[i]; k++)
            {
                if (j >= v[i][k])
                    dp[i][j] = max(dp[i][j], dp[i - 1][j - v[i][k]] + w[i][k]);
            }
        }
    }
    cout << dp[n][m] << endl;
    return 0;
}
```

#include <bits/stdc++.h>

using namespace std;

const int N = 1010;

int w[N][N], v[N][N], s[N];
int n, m;

int dp[N];
int main()
{

    {
        cin >> s[i];
        for (int j = 1; j <= s[i]; j++)
        {
            cin >> v[i][j] >> w[i][j];
        }
    }

    for (int i = 1; i <= n; i++)
    {
        for (int j = m; j >= 0; j--)
        {
            // dp[i][j] = dp[i - 1][j];
            for (int k = 0; k <= s[i]; k++)
            {
                if (j >= v[i][k])
                    dp[j] = max(dp[j], dp[j - v[i][k]] + w[i][k]);
            }
        }
    }
    cout << dp[m] << endl;
    return 0;
}
